高斯模糊

#include <opencv2/opencv.hpp>

using namespace cv;

void example4(const cv::Mat & image ){

    namedWindow("Example4-in",WINDOW_AUTOSIZE);
    namedWindow("Example4-out",WINDOW_AUTOSIZE);

    imshow("Example5-in", image);

    Mat out;

    GaussianBlur(image, out, Size(21,21), 3, 3);
    GaussianBlur(out, out, Size(21,21), 3, 3);

    imshow("Example4-out", out);

    waitKey(0);
}

int main(int argc, char** argv){

    Mat img = imread("/Users/caijiangnan/Desktop/Jojo.png");
    if(img.empty())return -1;

    example4(img);
    return 0;
}

#include <opencv2/opencv.hpp>

using namespace cv;

int main(int argc, char** argv){

    Mat img_rgb, img_gry, img_cny;
    namedWindow("Example Gray", WINDOW_AUTOSIZE);
    namedWindow("Example Canny", WINDOW_AUTOSIZE);

    img_rgb = imread("/Users/caijiangnan/Desktop/Jojo.png");

    cvtColor(img_rgb, img_gry, COLOR_BGR2GRAY);
    imshow("Example Gray", img_gry);

    Canny(img_gry, img_cny, 10, 100, 3, true);
    imshow("Example Canny", img_cny);

    waitKey(0);
    return 0;
}
